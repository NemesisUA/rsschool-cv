example: 132189 => 1 + 3 + 2 + 1 + 8 + 9 = 24 => 2 + 4 = 6

                
function digitalRoot(n) {
    return n < 10 ? n : digitalRoot(String(n).split('').reduce((a, b) => a + +b, 0));
}
